fn caml_string_equal(p1: &eq, p2: &eq) -> &eq {
  if p1 == p2 { return 1 }
  let s1 = p1 as &string;
  let s2 = p2 as &string;
  let len = array_len(s1);
  if len != array_len(s2) { return 0 }
  for (let i; i++; i <s len) {
    if unsigned(s1[i]) != unsigned(s2[i]) { return 0 }
  }
  1
}
